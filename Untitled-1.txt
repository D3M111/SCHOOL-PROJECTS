                desc: "A peaceful retreat away from the noise. Rolling hills, agriculture, and fresh air provide resources and relaxation for the city.",

        };

        // This is the Main Function triggered by the onclick Event
        function updateMap(locationKey, clickedButton) {
            
            // 1. Update the Data (Image and Text)
            const data = mapData[locationKey];
            const mapImage = document.getElementById('map-image');
            const title = document.getElementById('location-title');
            const desc = document.getElementById('location-desc');

            // Apply fade effect
            mapImage.style.opacity = 0;
            
            setTimeout(() => {
                mapImage.src = data.img;
                title.innerText = data.title;
                desc.innerText = data.desc;
                mapImage.style.opacity = 1;
            }, 200); // Wait 200ms for fade out before changing

            // 2. Update the Buttons (Visual Feedback)
            // Remove 'active' class from all buttons
            const allButtons = document.querySelectorAll('.map-btn');
            allButtons.forEach(btn => btn.classList.remove('active'));

            // Add 'active' class to the button that was just clicked
            clickedButton.classList.add('active');
        }
    </script>
